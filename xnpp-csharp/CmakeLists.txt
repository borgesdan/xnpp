cmake_minimum_required(VERSION 3.20)

set(PROJECT_NAME xnpp-csharp)
set(PROJECT_INCLUDES_DIR "${CMAKE_CURRENT_SOURCE_DIR}/include")

add_library(${PROJECT_NAME}
    "source/IO/BinaryReader.cpp"
    "source/IO/BinaryWriter.cpp"
    "source/IO/FileStream.cpp"
    "source/IO/MemoryStream.cpp"
    "source/IO/Stream.cpp"

    "source/Windows/Forms/Screen.cpp"
    "source/Windows/Forms/SystemInformation.cpp"
    
    "source/BitConverter.cpp"
    "source/Event.cpp"
    "source/Type.cpp"
    "source/Platform.cpp"
)

#SDL3
find_package(SDL3 CONFIG REQUIRED)

target_link_libraries(${PROJECT_NAME}       
    PRIVATE 
        SDL3::SDL3 
)

target_compile_features(${PROJECT_NAME} PRIVATE cxx_std_20)

target_include_directories(${PROJECT_NAME}
    PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:include>            
)

# Lógica mágica de exportação/importação:
if(BUILD_SHARED_LIBS)
    # Se for DLL, dizemos APENAS para o xnpp exportar os símbolos.
    # Os consumidores não recebem essa macro e caem no 'dllimport'.
    target_compile_definitions(${PROJECT_NAME} PRIVATE XNPP_EXPORT_DLL)
else()
    # Se for ESTÁTICO, nós desligamos o import/export.
    # Usamos PUBLIC para que qualquer target que faça link com xnpp 
    # (como seus .exe) também herde a macro XNPP_STATIC automaticamente!
    target_compile_definitions(${PROJECT_NAME} PUBLIC XNPP_STATIC)
endif()
